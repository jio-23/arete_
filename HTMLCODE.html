        class EatingDisorderRecoveryApp {

            constructor() {

                this.darkMode = false;

                this.profilePicURL = "";

                this.username = "";

                this.loadSettings(); // Load saved settings on startup

                this.loginPage();

            }

            loadSettings() {

                // Simulate loading settings from local storage or a similar mechanism

                const savedDarkMode = localStorage.getItem('darkMode');

                const savedUsername = localStorage.getItem('username');

                const savedProfilePicURL = localStorage.getItem('profilePicURL');

                if (savedDarkMode !== null) {

                    this.darkMode = JSON.parse(savedDarkMode);

                    this.toggleDarkMode(this.darkMode);

                }

                if (savedUsername) {

                    this.username = savedUsername;

                    document.getElementById("username").textContent = this.username;

                    document.getElementById("username").style.display = "block";

                }

                if (savedProfilePicURL) {

                    this.profilePicURL = savedProfilePicURL;

                    document.getElementById("profile-pic").src = this.profilePicURL;

                    document.getElementById("profile-pic").style.display = "block";

                }

            }

            saveSettings() {

                // Simulate saving settings to local storage

                localStorage.setItem('darkMode', JSON.stringify(this.darkMode));

                localStorage.setItem('username', this.username);

                localStorage.setItem('profilePicURL', this.profilePicURL);

            }

            loginPage() {

                const appContainer = document.getElementById("app");

                appContainer.innerHTML = `

                    <h1>Login</h1>

                    <label for="email">Email:</label>

                    <input type="email" id="email" placeholder="Enter your Gmail ID"><br>

                    <label for="password">Password:</label>

                    <input type="password" id="password" placeholder="Enter your password"><br>

                    <label for="usernameInput">Username:</label>

                    <input type="text" id="usernameInput" placeholder="Enter your username"><br>

                    <button id="loginBtn">Login</button>

                `;

                const loginBtn = document.getElementById("loginBtn");

                if (loginBtn) {

                    loginBtn.addEventListener("click", () => {

                        const email = document.getElementById("email").value;

                        const password = document.getElementById("password").value;

                        const username = document.getElementById("usernameInput").value;

                        if (email === "" || password === "" || username === "") {

                            alert("Please enter all details");

                            return;

                        }

                        this.username = username;

                        document.getElementById("username").textContent = username;

                        document.getElementById("username").style.display = "block";

                        this.saveSettings(); // Save username on login

                        this.page2();

                    });

                }

            }

                    <textarea id="notes" placeholder="Write your notes"></textarea><br>

                    <h2>Chat with Chatbot</h2>

                    <label for="chatMessage">Type your message:</label>

                    <input type="text" id="chatMessage" placeholder="Your message">

                    <button id="sendMessageBtn">Send Message</button>

                `;

                const sendMessageBtn = document.getElementById("sendMessageBtn");

                if (sendMessageBtn) {

                    sendMessageBtn.addEventListener("click", () => {

                        const message = document.getElementById("chatMessage").value;

                        document.getElementById("chatResponse").textContent = "Chatbot: " + message;

                        // In a real application, you would send this message to a backend

                    });

                }

                const settingsBtn = document.getElementById("settingsBtn");

                if (settingsBtn) {

                    settingsBtn.addEventListener("click", () => this.settings());

                }

            }

            settings() {

                const appContainer = document.getElementById("app");

                appContainer.innerHTML = `

                    <h1>Settings</h1>

                    <label for="darkModeToggle">Dark Mode:</label>

                    <input type="checkbox" id="darkModeToggle"><br>

                    <label for="profilePic">Upload Profile Picture:</label>

                    <input type="file" id="profilePic" accept="image/*"><br>

                    <button id="saveSettingsBtn">Save Settings</button>

                `;

                const darkModeToggle = document.getElementById("darkModeToggle");

                if (darkModeToggle) {

                    darkModeToggle.checked = this.darkMode;

                    darkModeToggle.addEventListener("change", (event) => {

                        this.toggleDarkMode(event.target.checked);

                        this.saveSettings(); // Save dark mode preference

                    });

                }

                const profilePicInput = document.getElementById("profilePic");

                if (profilePicInput) {

                    profilePicInput.addEventListener("change", (event) => {

                        this.uploadProfilePicture(event);

                    });

                }

                const saveSettingsBtn = document.getElementById("saveSettingsBtn");

                if (saveSettingsBtn) {

                    saveSettingsBtn.addEventListener("click", () => this.page2());

                }

            }

            uploadProfilePicture(event) {

                const file = event.target.files[0];

                if (file) {

                    const reader = new FileReader();

                    reader.onload = (e) => {

                        this.profilePicURL = e.target.result;

                        document.getElementById("profile-pic").src = this.profilePicURL;

                        document.getElementById("profile-pic").style.display = "block";

                        this.saveSettings(); // Save profile picture URL

                    };

                    reader.readAsDataURL(file);

                }

            }

            toggleDarkMode(enabled) {

                this.darkMode = enabled;

                if (this.darkMode) {

                    document.body.classList.add("dark-mode");

                } else {

                    document.body.classList.remove("dark-mode");

                }

            }

        }
